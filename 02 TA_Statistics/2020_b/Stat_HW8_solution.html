def r_ttest(r, n, alpha):
    t_value = r * ((n-2)/(1 - r**2)) ** 0.5
    LCL = stats.t.ppf(alpha/2, n - 2)
    UCL = stats.t.ppf(1-alpha/2, n - 2)
    print('t = ', t_value)
    print('t_LCL = ', LCL)
    print('t_UCL = ', UCL)
    return t_value, LCL, UCL
